#Signing the White-Gold Concordat
country_event = {
	id = great_war.6
	title = great_war.6.t
	desc = great_war.6.d
	
	is_triggered_only = yes
	
	option = {
	    name = great_war.6.a
		add_ideas = { idea_white_gold_concordat }
		hidden_effect = {
			news_event = great_war.8
		}
		if = {
			limit = { HAM = { is_subject_of = CYR } }
			ALD = { 
				transfer_state = 124
				transfer_state = 35
				transfer_state = 32
				transfer_state = 38
				transfer_state = 198
				transfer_state = 37
				transfer_state = 36
				transfer_state = 43
			}
		}
		every_country = {
			limit = { is_subject_of = CYR has_idea = nine_divines }
			add_ideas = eight_divines
		}
	}
}
