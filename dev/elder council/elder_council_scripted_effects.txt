recalc_elder_council_icon_frames = {
	
	set_temp_variable = { elder_council_temp = elder_council_support }
	divide_temp_variable = { elder_council_temp = elder_council_total }
	multiply_temp_variable = { elder_council_temp = 21 }
	if = {
		limit = { check_variable = { elder_council_temp > 0 } }
		clamp_temp_variable = {
			var = elder_council_temp
			min = 1
			max = 21
		}
	}
	else = {
		set_temp_variable = { elder_council_temp = 1 }
	}
	set_variable = { elder_council_icon_frame = elder_council_temp }
}
