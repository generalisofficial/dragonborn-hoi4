	form_elsweyr = {

		icon = generic_form_nation

		available = {
			all_state = { 
				OR = { 
					NOT = { is_on_continent = elsweyr }
					is_owned_by = ROOT
				}
			}
		}

			OR = {
				original_tag = ANQ
				original_tag = PEL

			set_cosmetic_tag = ELS
