lar_local_recruitment = {
	#unlocks decisions to recruit in nirn
	recruit_in_nirn = {

		#icon = generic_research

		available = {
			
		}

		visible = {
			capital_scope = {
				NOT = {
					is_on_continent = nirn
				}
			}
			OR = {
				any_owned_state = {
					is_on_continent = nirn
				}
				any_allied_country = {
					any_owned_state = {
						is_on_continent = nirn
					}
				}
				any_other_country = {
					is_subject_of = ROOT
					any_owned_state = {
						is_on_continent = nirn
					}
				}
			}
		}

		fire_only_once = yes
		cost = 50

		modifier = {
			civilian_factory_use = 3
		}

		days_remove = 90

		remove_effect = {
			set_country_flag = nirn_recruitment_unlocked
		}

		ai_will_do = {
			base = 0
		}
	}

	recruit_in_nirn_state = {

		icon = onmap_recruit_operative

		available = {
			
		}

		state_target = yes
		target_trigger = {
			FROM = {
				is_on_continent = nirn
				CONTROLLER = {
					NOT = {
						tag = ROOT
					}
				}
				is_capital = yes
			}
		}
		visible = {
			OR = {
				has_country_flag = nirn_recruitment_unlocked
				capital_scope = { is_on_continent = nirn }
			}
			NOT = { has_country_flag = nirn_recruitment_in_process }
		}

		fire_only_once = no
		cost = 0

		days_re_enable = 180
		on_map_mode = map_only
		complete_effect = {
			set_country_flag = { flag = nirn_recruitment_in_process days = 180 value = 1 }
			if = {
				limit = {
					num_of_operatives < num_operative_slots
				}
				custom_effect_tooltip = recruit_in_state_tt
			}
			else = {
				custom_effect_tooltip = recruit_in_state_to_pool_tt
			}
			hidden_effect = {
				create_operative_leader = {
					bypass_recruitment = yes
					nationalities = { FROM.OWNER }
					portrait_tag_override = FROM.OWNER
				}
			}
		}

		ai_will_do = {
			base = 0
		}
	}
}
